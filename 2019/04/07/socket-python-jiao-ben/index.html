1.socket通信脚本

#server端脚本

#! /usr/bin/python3.6
# coding=utf-8
import socket
from datetime import datetime

HOST="127.0.0.1"
PORT=1234
BUFSIZE=1024
ADDR=(HOST,PORT)

s=socket.socket(socket.AF_INET,socket.SOCK_STREAM)
s.bind(ADDR)
s.listen(1)                                            #listen函数传入允许连接的客户端的数量
print("waiting for connct...")
conn,addr=s.accept()
print("conncting from : ",addr)
while True:
    data=conn.recv(BUFSIZE)
    data=str(data,encoding="utf-8")
    #data=data.decode("utf-8")
    if data=="exit":
        break
    if data:
        print("client(%s):%s"%(str(datetime.utcnow()),data))
    data=input("server"+"("+str(datetime.utcnow())+")"+":")
    if data=="exit":
        break
    if data:
        conn.send(data.encode("utf-8"))
conn.close()
s.close()

1.从上面可以看出服务端首先绑定一个ip和端口，然后监听该IP和端口，通过accept函数记录连接客户端的信息，再通过send函数和recv函数实现数据交流，其中需要注意的一点是发送信息的编码问题，上面可以看出都是利用utf-8进行传输的。
2.对比服务端和客户端还可以发现，客户端是直接利用创建的socket与服务端进行交流的，而服务端则是利用建立的连接与特定的客户端交流

#server

#! /usr/bin/python3.6
# coding=utf-8
import socket
import os
BUFSIZE=1024
HOST="127.0.0.1"
PORT=6666
ADDR=(HOST,PORT)
def sendfile(conn):
    str=conn.recv(BUFSIZE)
    filename=str.decode("utf-8")
    if filename:
        print("the client request my file:",filename)
        if os.path.exists(filename):
            print("i have such file,begin to deliver!")
            conn.send(b'yes')
            #conn.recv(BUFSIZE)
            with open(filename,"rb") as f:
                while True:
                    data=f.read(BUFSIZE)
                    conn.send(data)
                    if len(data)<BUFSIZE:
                        break
            print("%s is deliver successfully."%filename)
        else:
            print("Sorry,i have no such file")
            conn.send(b'no')
s=socket.socket(socket.AF_INET,socket.SOCK_STREAM)
s.bind(ADDR)
s.listen(1)
print("wait for connecting.")
while True:
    conn,addr=s.accept()
    sendfile(conn)

4.UDP通信

#server
#! /usr/bin/python3.6
# coding=utf-8
import socket
import psutil
HOST="127.0.0.1"
PORT=2022
BUFSIZE=1024
ADDR=(HOST,PORT)
def do_cpu():
    data=str(psutil.cpu_percent(0))+'%\n'
    count=0
    for process in psutil.process_iter():
        data=data+process.name()
        data=data+','+str(process.pid)
        cpu_usage_rate_process=str(process.cpu_percent(0))+'%\n'
        count+=1
        data=data+','+cpu_usage_rate_process
        if count==10:
            break
    return data
s=socket.socket(socket.AF_INET,socket.SOCK_DGRAM)
s.bind(ADDR)
print("bind udp on 2022")
while True:
    (info,addr)=s.recvfrom(BUFSIZE)
    data=do_cpu()
    s.sendto(data.encode("utf-8"),addr)
    print("the client is ",addr)
    print("sended cpu data is :",data)

5.ICMP报文

#! /usr/bin/python3.6
# coding=utf-8
import struct
import array
import time
import socket
def checksum(packet):
    if len(packet)&1:
        packet=packet+'\0'
    words=array.array('h',packet)
    sum=0
    for word in words:
        sum+=(word&0xffff)
    sum=(sum>>16)+(sum&0xffff)
    sum+=(sum>>16)
    return (~sum)&0xffff
header=struct.pack("bbHHh",8,0,0,1234,5)
data=struct.pack('d',time.time())
packet=header+data
chksum=checksum(packet)
header=struct.pack("bbHHh",8,0,chksum,1234,5)
packet=header+data
s=socket.socket(socket.AF_INET,socket.SOCK_RAW,socket.getprotobyname("icmp"))
s.settimeout(3)
ip=input("please input an ip:")
for kk in range(4):
    try:
        t1=time.time()
        s.sendto(packet,(ip,0))
        (r_data,r_addr)=s.recvfrom(1024)
        t2=time.time()
    except Exception as e:
        print("Error is ",e)
        continue
    print("receive the response from %s ,data is %d bytes,time is %.2f ms"%(r_addr[0],len(r_data),(t2-t1)*1000))
    h1,h2,h3,h4,h5=struct.unpack("bbHHh",r_data[20:28])
    print("type=%d,code=%d,checksum=%u,id=%u,sn=%d"%(h1,h2,h3,h4,h5))


这里手写了一个ping工具，由上图可以看出icmp报文由类型（0或8代表：回应请求），代码占8位表示发送报文的原因，校验和占十六位，标识符和序号占32字节，数据部分长度可变。上述代码中每一层加一个header，从而构造出数据包。

from multiprocessing import Process
import socket
import os
def sendfile(conn):
    str1=conn.recv(1024)
    filename=str1.decode("utf-8")
    print("i am a chirld process ,my id is %d"%os.getpid())
    if os.path.exists(filename):
        print("i have the file")
        conn.send(b'yes')
        conn.recv(1024)
        size=1024
        with open(filename,'rb') as f:
            while True:
                data=f.read(size)
                conn.send(data)
                if len(data)<size:
                    break
        print("the file is download")

        print("sorry,the file is none")
        conn.send(b'no')
    conn.close()

s.listen(100)
print("i am parent process ,my id is %d"%os.getpid())
print("waiting for connecting....")

    conn,addr=s.accept()
    p=Process(target=sendfile,args=(conn,))
    p.start()
