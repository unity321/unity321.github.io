class Repeat
    @@total=0
    def initialize(string,times)
        @string=string
        @times=times
    end
    def repeat
        @@total+=@times
        return @string*@times
    end
    def total
        "Total tiems , so far: " + @@total.to_s
    end
end
data=Repeat.new("hello ",5)
data1=Repeat.new("ruby ",2)
p data.repeat
p data.total
p data1.repeat
p data1.total

模块
module Dice
    def roll
        r_1=rand(6)
        r_2=rand(6)
        r1=r_1>0?r_1:1
        r2=r_2>0?r_2:6
        total=r1+r2
        printf "Your rolled %d and %d (%d).\n",r1,r2,total
        total
    end
end
class Game
    include Dice
end
g=Game.new
g.roll
#模块可以实现多继承类似go的interface
可见度
public                #方法可以被定义它的类和子类访问，并可以被类和子类的实例对象调用；
protected            #方法可以被定义它的类和子类访问，不能被类和子类的实例对象调用，但可以被该类的实例对象(所有)访问；
private                #方法可以被定义它的类和子类访问，不能被类和子类的实例对象调用，且实例对象只能访问自己的private方法。

#以上的陈述中，请注意“调用”和“访问”的区别。



class Game
    def name(name)
        @name=name
        p @name
    end
    private
    def add(a,b)
        p a+b
    end
    def div(a,b)
        p a/b
    end
    protected
    def add1(a,b)
        p a+b
    end
    def div1(a,b)
        p a/b
    end

    public
    def get
        p add(1,2)
        p div(2,1)
        p add1(1,2)
        p div1(2,1)
    end


end
